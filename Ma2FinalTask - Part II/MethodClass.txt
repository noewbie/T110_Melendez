package Ma2FinalTaskPart_2
//Parent Class
public class MethodClass{
	
	public double triangleArea(double base, double height) {
		return (base * height) / 2;
	}
	
	public double rectangleArea(double legnth, double height) {
		return legnth * height;
	}
	
	public double parrallelogramArea(double base, double height) {
		return base * height;
	}
	
	public double circleArea(double radius) {
		return 3.14 * radius * radius;
	}
}
